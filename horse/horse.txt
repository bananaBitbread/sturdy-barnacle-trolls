    <script type="text/javascript">
      window.onload = function() {
        const legsDiv = document.getElementById('legs');
        const legSegment = "               | | | |             || |  '\n";

        function addLegs() {
          // Creating a new pre for each batch keeps them perfectly stacked
          const newBatch = document.createElement('pre');
          newBatch.textContent = legSegment.repeat(25);
          legsDiv.appendChild(newBatch);
        }

        window.onscroll = function() {
          if ((window.innerHeight + window.scrollY) >= document.body.offsetHeight - 1000) {
            addLegs();
          }
        };

        addLegs(); // Start the infinite growth
      };
    </script>
